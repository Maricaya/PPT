% ==> Packages <== %

\usepackage{titlesec}

% ==> Fonts <== %

    ItalicFont={PortoSans-LightItalic.otf},
    BoldItalicFont={PortoSans-RegularItalic.otf}

    Path=fonts/portosans/,
